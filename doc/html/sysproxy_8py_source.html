732 def get_sys_proxy_parsed(self, proxy='http_proxy') -> Proxyconf:
733 _proxy = self.get_sys_http_proxy() if proxy == 'http_proxy' else self.get_sys_env(proxy)
734 if (_proxy is None) or not (_proxy['user'] or _proxy['system']):
735 return None
736 # prefer user (local) proxy config over system
737 _proxy = _proxy['user'] or _proxy['system']
738 spl = _proxy.split('://')
739 prot = spl[0].lower() if len(spl) > 1 else 'http'
740 other = spl[1] if len(spl) > 1 else _proxy
741 spl1 = other.split('@')
742 if len(spl1) > 1:
743 spl11 = spl1[0].split(':')
744 uname = spl11[0]
745 passw = spl11[1] if len(spl11) > 1 else None
746 else:
747 uname, passw = (None, None)
748 other2 = spl1[1] if len(spl1) > 1 else other
749 spl2 = other2.split(':')
750 host = spl2[0]
751 port = int(spl2[1]) if len(spl2) > 1 else None
752 return Proxyconf(None, prot, host or '', port or 3128, not uname is None, uname or '', passw or '')
