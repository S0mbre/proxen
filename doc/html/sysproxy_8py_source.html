429 if OS != 'Windows':

431 if isinstance(branch, str):
432 branch = WIN_REG_BRANCHES[branch]
433 if keyname == WIN_ENV_LOCAL_KEY and valname == WIN_DUMMY_KEYNAME:
434 return ('t', winreg.REG_SZ)
435 k = None
436 res = None
437 try:
438 k = winreg.OpenKeyEx(branch, keyname, 0, winreg.KEY_ALL_ACCESS)
439 try:
440 val = winreg.QueryValueEx(k, valname)

578 def win_list_reg(self, keyname, branch='HKEY_CURRENT_USER', expand_vars=True, with_types=False) -> dict:
579 if OS != 'Windows':
580 raise Exception('This method is available only on Windows platforms!')
581 if isinstance(branch, str):
582 branch = WIN_REG_BRANCHES[branch]
583 res = {}
584 if expand_vars:
585 reg = re.compile(r'%(.+?)%')
586 try:
587 k = winreg.OpenKeyEx(branch, keyname, 0, winreg.KEY_READ)
588 info = winreg.QueryInfoKey(k)
589 for i in range(info[1]):
590 try:
591 val = winreg.EnumValue(k, i)
592 if val[0] == WIN_DUMMY_KEYNAME:
593 continue
594 if not expand_vars or not isinstance(val[1], str):
595 res[val[0]] = val[1:] if with_types else val[1]
596 else:
597 v = reg.sub(lambda m: os.environ.get(m[1], m[0]), val[1])
598 res[val[0]] = (v, val[2]) if with_types else v
599 except OSError:
600 break
601 except:
602 traceback.print_exc()
603 finally:
604 if k: winreg.CloseKey(k)
605 return res

811 spl = _proxy.split('://')
812 prot = spl[0].lower() if len(spl) > 1 else 'http'
813 other = spl[1] if len(spl) > 1 else _proxy
814 spl1 = other.split('@')
